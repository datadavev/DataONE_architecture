@startuml
title Interactions 02 - Client query to get list of GUIDs from metadata \nsearch 
actor User
participant "Client" as app_client << Application >>
User -> app_client
participant "Authorization API" as m_authorize << Member Node >>
app_client -> m_authorize: login (user, pw)
app_client <- m_authorize: token or failure
participant "Query API" as c_query << Coordinating Node >>
app_client -> c_query: query (token, query)
participant "Authorization API" as c_authorize << Coordinating Node >>
c_query -> c_authorize: isAuth (token, resultset)
participant "Verify API" as c_ver << Coordinating Node >>
c_authorize -> c_ver: isValidToken (token)
c_ver -> c_authorize: T or F
c_authorize -> c_query: T or F
c_query -> app_client: resultset
@enduml
