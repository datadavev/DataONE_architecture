@startuml
title Interactions: 17 - All CRUD logs are aggregated at Coordinating Nodes
note left
Assume Coordinating Node has authority for accessing Member Node log records.
end note
participant "CRUD API" as m_crud_a << Member Node A >>
participant "Synchronization API" as c_sync << Coordinating Node >>
c_sync -> m_crud_a: getLogRecords (fromDate, toDate)
c_sync <-- m_crud_a: records or fail
participant "CRUD API" as m_crud_b << Member Node B >>
c_sync -> m_crud_b: getLogRecords (fromDate, toDate)
c_sync <-- m_crud_b: records or fail
@enduml
